rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null &&
             get(/databases/$(database)/documents/users/$(request.auth.uid)).data.role == "admin";
    }

    // USERS collection
    match /users/{userId} {

      // Admin full access to create/edit/delete any user
      allow read, write: if isAdmin();

      // Employees can only read their own profile
      allow read: if request.auth != null && request.auth.uid == userId;
    }

    // TASKS collection
    match /tasks/{taskId} {

      // Admin full access
      allow read, write: if isAdmin();

      // Employee read only their assigned tasks
      allow read: if request.auth != null &&
                   resource.data.assignedTo == request.auth.uid;

      // Employee update only their own task
      allow update: if request.auth != null &&
                     request.resource.data.assignedTo == request.auth.uid;
    }
  }
}
